start_file=start_x.elf
fixup_file=fixup_x.dat

# Kernel location
kernel=zImage

# How much memory in MB to assign to the GPU. Camera firmware (MMAL)
# needs headroom for capture buffers; 64 was too tight on this kernel
# (stream opened but never delivered a frame). 128 matches what most
# Pi-Zero camera projects use.
gpu_mem=128

# Enable Camera module. `start_x=1` loads the GPU camera firmware, which
# drives the OV5647 sensor and exposes /dev/video0 via the in-kernel
# bcm2835-v4l2 shim. Do NOT also set dtoverlay=ov5647 — that binds the
# sensor to the kernel unicam subdev and fights the firmware for CSI/I2C
# ownership (symptom: /dev/video0 opens but never produces frames).
start_x=1

# Enable SPI
dtparam=spi=on

# Set the bootloader delay to 0 seconds. The default is 1s if not specified
boot_delay=0
disable_splash=1
force_turbo=1
